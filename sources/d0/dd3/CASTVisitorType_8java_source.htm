111  public CASTVisitorType( @Nonnull final Set<IAction> p_actions, @Nonnull final Set<IRule> p_rules )
112  {
113  m_actions = p_actions.stream().collect( Collectors.toMap( i -> i.name(), i -> i ) );
114  p_rules.stream().forEach( i -> m_rules.put( i.identifier().fqnfunctor(), i ) );

169  @Override
170  public final Object visitUnification( final TypeParser.UnificationContext p_context )
171  {
172  final Object l_constraint = this.visitUnification_constraint( p_context.unification_constraint() );
173 
174  if ( l_constraint instanceof IExpression )
175  return new CExpressionUnify(
176  p_context.AT() != null,
177  (ILiteral) this.visitLiteral( p_context.literal() ),
178  (IExpression) l_constraint
179  );
180 
181  if ( l_constraint instanceof IVariable<?> )
182  return new CVariableUnify(
183  p_context.AT() != null,
184  (ILiteral) this.visitLiteral( p_context.literal() ),
185  (IVariable<?>) l_constraint
186  );
187 
188  return new CDefaultUnify( p_context.AT() != null, (ILiteral) this.visitLiteral( p_context.literal() ) );
189  }

269  @Override

271  {
272  if ( ( p_context == null ) || ( p_context.isEmpty() ) )
273  return Collections.<ITerm>emptyList();
274

276  .map( i -> this.visitTerm( i ) )
