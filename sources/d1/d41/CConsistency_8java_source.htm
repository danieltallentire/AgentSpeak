134  @Override
135  public final IConsistency call() throws Exception
136  {
137  if ( m_data.size() < 2 )
138  return this;
139 
140  // get key list of map for addressing elements in the correct order
141  final ArrayList<IAgent<?>> l_keys = new ArrayList<>( m_data.keySet() );
142 
143  // calculate markov chain transition matrix
144  final DoubleMatrix2D l_matrix = new DenseDoubleMatrix2D( m_data.size(), m_data.size() );
145  IntStream.range( 0, l_keys.size() )
146  .parallel()
147  .boxed()
148  .forEach( i ->
149  {
150  final IAgent<?> l_item = l_keys.get( i );
151  IntStream.range( i + 1, l_keys.size() )
152  .boxed()
153  .forEach( j ->
154  {
155  final double l_value = this.getMetricValue( l_item, l_keys.get( j ) );
156  l_matrix.setQuick( i, j, l_value );
157  l_matrix.setQuick( j, i, l_value );
158  } );
159 
160  // row-wise normalization for getting probabilities
161  final double l_norm = Algebra.DEFAULT.norm1( l_matrix.viewRow( i ) );
162  if ( l_norm != 0 )
163  l_matrix.viewRow( i ).assign( Mult.div( l_norm ) );
164 
165  // set epsilon slope for preventing periodic markov chains
166  l_matrix.setQuick( i, i, m_epsilon );
167  } );
168 
169  // check for a zero-matrix
170  final DoubleMatrix1D l_eigenvector = l_matrix.zSum() <= m_data.size() * m_epsilon
171  ? new DenseDoubleMatrix1D( m_data.size() )
172  : m_algorithm.getStationaryDistribution( m_iteration, l_matrix );
173 
174  // calculate the inverted probability and normalize with 1-norm
175  l_eigenvector.assign( PROBABILITYINVERT );
176  l_eigenvector.assign( Functions.div( Algebra.DEFAULT.norm1( l_eigenvector ) ) );
177 
178  // set consistency value for each entry and update statistic
179  m_statistic.clear();
180  IntStream.range( 0, l_keys.size() )
181  .boxed()
182  .forEach( i ->
183  {
184  m_statistic.addValue( l_eigenvector.get( i ) );
185  m_data.put( l_keys.get( i ), l_eigenvector.get( i ) );
186  } );
187 
188  return this;
189  }

329  public final DoubleMatrix1D getStationaryDistribution( final int p_iteration, final DoubleMatrix2D p_matrix )
330  {
331  final DoubleMatrix1D l_eigenvector;
332  switch ( this )
333  {
334  case FIXPOINT:
335  l_eigenvector = getLargestEigenvector( p_matrix, p_iteration );
336  break;
337 
338  case NUMERICAL:
339  l_eigenvector = getLargestEigenvector( p_matrix );
340  break;
341 
342  default:
343  throw new CIllegalStateException( CCommon.languagestring( this, "algorithm", this ) );
344  }
345 
346  // normalize eigenvector and create positiv oriantation
347  l_eigenvector.assign( Mult.div( Algebra.DEFAULT.norm1( l_eigenvector ) ) );
348  l_eigenvector.assign( Functions.abs );
349 
350  return l_eigenvector;
351  }

362  private static DoubleMatrix1D getLargestEigenvector( final DoubleMatrix2D p_matrix, final int p_iteration )
363  {
364  final DoubleMatrix1D l_probability = DoubleFactory1D.dense.random( p_matrix.rows() );
365  IntStream.range( 0, p_iteration )
366  .forEach( i ->
367  {
368  l_probability.assign( Algebra.DEFAULT.mult( p_matrix, l_probability ) );
369  l_probability.assign( Mult.div( Algebra.DEFAULT.norm2( l_probability ) ) );
370  } );
371  return l_probability;
372  }

380  private static DoubleMatrix1D getLargestEigenvector( final DoubleMatrix2D p_matrix )
381  {
382  final EigenvalueDecomposition l_eigen = new EigenvalueDecomposition( p_matrix );
383 
384  // gets the position of the largest eigenvalue in parallel and returns the eigenvector
385  final double[] l_eigenvalues = l_eigen.getRealEigenvalues().toArray();
386  return l_eigen.getV().viewColumn(
387  IntStream.range( 0, l_eigenvalues.length - 1 ).parallel()
388  .reduce( ( i, j ) -> l_eigenvalues[i] < l_eigenvalues[j] ? j : i ).orElse( 0 )
389  );
390  }
