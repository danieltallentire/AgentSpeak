72  @Nonnull
73  @SuppressWarnings( "unchecked" )
74  protected static IFuzzyValue<Boolean> execute( final boolean p_parallel, @Nonnull final IContext p_context, @Nonnull final ILiteral p_value
75  )
76  {
77  // read current rules, if not exists execution fails
78  final Collection<IRule> l_rules = p_context.agent().rules().get( p_value.fqnfunctor() );
79  if ( l_rules == null )
80  return CFuzzyValue.from( false );
81 
82  // first step is the unification of the caller literal, so variables will be set from the current execution context
83  final ILiteral l_unified = p_value.allocate( p_context );
84 
85  // second step execute backtracking rules sequential / parallel
86  return (
87  p_parallel
88  ? l_rules.parallelStream()
89  : l_rules.stream()
90  ).map( i ->
91  {
92 
93  // instantiate variables by unification of the rule literal
94  final Set<IVariable<?>> l_variables = p_context.agent().unifier().unify( l_unified, i.identifier() );
95 
96  // execute rule
97  final IFuzzyValue<Boolean> l_return = i.execute(
98  false, i.instantiate( p_context.agent(), l_variables.stream() ),
99  Collections.<ITerm>emptyList(),
100  Collections.<ITerm>emptyList()
101  );
102 
103  // create rule result with fuzzy- and defuzzificated value and instantiate variable set
104  return new ImmutableTriple<>( p_context.agent().fuzzy().getValue().defuzzify( l_return ), l_return, l_variables );
105 
106  } )
107 
108  // find successfully ended rule
109  .filter( ImmutableTriple::getLeft )
110  .findFirst()
111 
112  // realocate rule instantiated variables back to execution context
113  .map( i ->
114  {
115 
116  i.getRight().parallelStream()
117  .filter( j -> j instanceof IRelocateVariable )
118  .forEach( j -> ( (IRelocateVariable) j ).relocate() );
119 
120  return i.getMiddle();
121 
122  } )
123 
124  // otherwise rule fails (default behaviour)
125  .orElse( CFuzzyValue.from( false ) );
126  }
