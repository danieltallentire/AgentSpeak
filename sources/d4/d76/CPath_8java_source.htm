420  private synchronized void normalize()
421  {
422  if ( m_path.isEmpty() )
423  return;
424 
425  // create path-copy and nomalize (remove dot, double-dot and empty values)
426  final List<String> l_dotremove = m_path.stream()
427  .filter( i -> ( i != null ) && ( !i.isEmpty() ) && ( !".".equals( i ) ) )
428  .collect( Collectors.toList() );
429  if ( l_dotremove.isEmpty() )
430  return;
431 
432  final String l_last = l_dotremove.get( l_dotremove.size() - 1 );
433  final List<String> l_backremove = IntStream.range( 0, l_dotremove.size() - 1 )
434  .boxed()
435  .filter( i -> !l_dotremove.get( i + 1 ).equals( ".." ) )
436  .map( l_dotremove::get )
437  .collect( Collectors.toList() );
438  if ( !"..".equals( l_last ) )
439  l_backremove.add( l_last );
440 
441  // clear internal path and add optimized path
442  m_path.clear();
443  m_path.addAll( l_backremove );
444  }
