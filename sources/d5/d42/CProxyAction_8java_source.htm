202  CActionWrapper( @Nonnull final ILiteral p_literal, @Nonnull final Map<IPath, IAction> p_actions )
203  {
204  // check parallel and inner execution
205  m_parallel = p_literal.hasAt();
206 
207 
208  // resolve action
209  m_action = p_actions.get( p_literal.fqnfunctor() );
210  if ( m_action == null )
211  throw new CIllegalArgumentException( org.lightjason.agentspeak.common.CCommon.languagestring( this, "actionunknown", p_literal ) );
212 
213  // check number of arguments and add action to the score cache
214  if ( p_literal.orderedvalues().count() < m_action.minimalArgumentNumber() )
215  throw new CIllegalArgumentException(
216  org.lightjason.agentspeak.common.CCommon.languagestring( this, "argumentnumber", p_literal, m_action.minimalArgumentNumber() ) );
217 
218  // resolve action arguments
219  m_arguments = Collections.unmodifiableMap( this.createSubExecutions( p_literal.orderedvalues().collect( Collectors.toList() ), p_actions ) );
220  }

267  @Nonnull
268  private Map<Integer, IExecution> createSubExecutions( @Nonnull final Collection<? extends ITerm> p_elements,
269  @Nonnull final Map<IPath, IAction> p_actions )
270  {
271  // convert collection to list and build map with indices
272  final List<? extends ITerm> l_elements = new LinkedList<>( p_elements );
273  return IntStream.range( 0, l_elements.size() )
274  .boxed()
275  .collect(
276  Collectors.toMap(
277  i -> i,
278  i ->
279  {
280  final ITerm l_term = l_elements.get( i );
281  return l_term instanceof ILiteral
282  ? new CActionWrapper( l_term.term(), p_actions )
283  : new CTermWrapper<>( l_term );
284  }
285  )
286  );
287  }

296  @Nonnull
297  private List<ITerm> subexecute( @Nonnull final IContext p_context, @Nonnull final Map<Integer, IExecution> p_execution )
298  {
299  return Collections.unmodifiableList( CCommon.replaceFromContext(
300  p_context,
301  ( m_parallel
302  ? p_execution.entrySet().parallelStream()
303  : p_execution.entrySet()
304  .stream() )
305  .flatMap( i ->
306  {
307  final List<ITerm> l_return = new LinkedList<>();
308  i.getValue().execute( m_parallel, p_context, Collections.emptyList(), l_return );
309  return l_return.stream();
310  } )
311  .collect( Collectors.toList() )
312  ) );
313  }
