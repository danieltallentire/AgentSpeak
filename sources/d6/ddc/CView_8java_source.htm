278  @SuppressWarnings( "varargs" )
279  private Stream<IView> walkdown( @Nonnull final IPath p_path, @Nullable final IViewGenerator... p_generator )
280  {
281  if ( p_path.empty() )
282  return Stream.of( this );
283 
284  final IView l_view;
285  final String l_root = p_path.get( 0 );
286 
287  synchronized ( this )
288  {
289  // add is run here for avoid overwriting view with a new object reference
290  l_view = m_beliefbase.viewOrDefault(
291  l_root,
292 
293  p_generator == null || p_generator.length == 0
294  ? null
295  : p_generator[0].apply( l_root, this )
296  );
297 
298  if ( l_view == null )
299  return Stream.empty();
300  m_beliefbase.add( l_view );
301  }
302 
303  return Stream.concat(
304  Stream.of( this ),
305  l_view.walk( p_path.subpath( 1 ), p_generator )
306  );
307  }
