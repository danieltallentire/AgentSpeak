336  public static double levenshtein( @Nonnull final String p_first, @Nonnull final String p_second, final double p_insertweight,
337  final double p_replaceweight, final double p_deleteweight )
338  {
339  // the array of distances
340  double[] l_cost = IntStream.range( 0, p_first.length() + 1 ).mapToDouble( i -> i ).toArray();
341  double[] l_newcost = new double[l_cost.length];
342 
343  for ( int j = 1; j < p_second.length() + 1; j++ )
344  {
345  l_newcost[0] = j;
346 
347  // calculate cost of operation for all characters
348  for ( int i = 1; i < l_cost.length; i++ )
349  l_newcost[i] = min(
350  l_cost[i - 1] + ( p_first.charAt( i - 1 ) == p_second.charAt( j - 1 ) ? 0 : p_replaceweight ),
351  l_newcost[i - 1] + p_deleteweight,
352  l_cost[i] + p_insertweight
353  );
354 
355  final double[] l_swap = l_cost;
356  l_cost = l_newcost;
357  l_newcost = l_swap;
358  }
359 
360  return l_cost[p_first.length()];
361  }

386  public static double ncd( @Nonnull final ECompression p_compression, @Nonnull final String p_first, @Nonnull final String p_second )
387  {
388  if ( p_first.equals( p_second ) )
389  return 0;
390 
391  final double l_first = compress( p_compression, p_first );
392  final double l_second = compress( p_compression, p_second );
393  return ( compress( p_compression, p_first + p_second ) - Math.min( l_first, l_second ) ) / Math.max( l_first, l_second );
394  }

405  private static double compress( @Nonnull final ECompression p_compression, @Nonnull final String p_input )
406  {
407  final DataOutputStream l_counting = new DataOutputStream( new NullOutputStream() );
408 
409  try (
410  final InputStream l_input = new ByteArrayInputStream( p_input.getBytes( StandardCharsets.UTF_8 ) );
411  final OutputStream l_compress = p_compression.get( l_counting )
412  )
413  {
414  IOUtils.copy( l_input, l_compress );
415  }
416  catch ( final IOException l_exception )
417  {
418  return 0;
419  }
420 
421  return l_counting.size();
422  }
