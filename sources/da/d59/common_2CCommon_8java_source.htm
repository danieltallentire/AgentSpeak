151  @Nonnull
152  public static Stream<IAction> actionsFromPackage( @Nullable final String... p_package )
153  {
154  return ( ( p_package == null ) || ( p_package.length == 0 )
155  ? Stream.of( MessageFormat.format( "{0}.{1}", PACKAGEROOT, "action.builtin" ) )
156  : Arrays.stream( p_package ) )
157  .flatMap( j ->
158  {
159  try
160  {
161  return ClassPath.from( Thread.currentThread().getContextClassLoader() )
162  .getTopLevelClassesRecursive( j )
163  .parallelStream()
164  .map( ClassPath.ClassInfo::load )
165  .filter( i -> !Modifier.isAbstract( i.getModifiers() ) )
166  .filter( i -> !Modifier.isInterface( i.getModifiers() ) )
167  .filter( i -> Modifier.isPublic( i.getModifiers() ) )
168  .filter( IAction.class::isAssignableFrom )
169  .map( i ->
170  {
171  try
172  {
173  return (IAction) i.newInstance();
174  }
175  catch ( final IllegalAccessException | InstantiationException l_exception )
176  {
177  LOGGER.warning( CCommon.languagestring( CCommon.class, "actioninstantiate", i, l_exception ) );
178  return null;
179  }
180  } )
181 
182  // action can be instantiate
183  .filter( Objects::nonNull )
184 
185  // check usable action name
186  .filter( CCommon::actionusable );
187  }
188  catch ( final IOException l_exception )
189  {
190  throw new UncheckedIOException( l_exception );
191  }
192  } );
193  }

239  private static boolean actionusable( final IAction p_action )
240  {
241  if ( ( p_action.name().empty() ) || ( p_action.name().get( 0 ).trim().isEmpty() ) )
242  {
243  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionnameempty" ) );
244  return false;
245  }
246 
247  if ( !Character.isLetter( p_action.name().get( 0 ).charAt( 0 ) ) )
248  {
249  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionletter", p_action ) );
250  return false;
251  }
252 
253  if ( !Character.isLowerCase( p_action.name().get( 0 ).charAt( 0 ) ) )
254  {
255  LOGGER.warning( CCommon.languagestring( CCommon.class, "actionlowercase", p_action ) );
256  return false;
257  }
258 
259  return true;
260  }

270  @Nonnull
271  private static Stream<Method> methods( final Class<?> p_class )
272  {
273  final Pair<Boolean, IAgentAction.EAccess> l_classannotation = CCommon.isActionClass( p_class );
274  if ( !l_classannotation.getLeft() )
275  return p_class.getSuperclass() == null
276  ? Stream.empty()
277  : methods( p_class.getSuperclass() );
278 
279  final Predicate<Method> l_filter = IAgentAction.EAccess.WHITELIST.equals( l_classannotation.getRight() )
280  ? i -> !CCommon.isActionFiltered( i, p_class )
281  : i -> CCommon.isActionFiltered( i, p_class );
282 
283  return Stream.concat(
284  Arrays.stream( p_class.getDeclaredMethods() )
285  .parallel()
286  .peek( i -> i.setAccessible( true ) )
287  .filter( i -> !Modifier.isAbstract( i.getModifiers() ) )
288  .filter( i -> !Modifier.isInterface( i.getModifiers() ) )
289  .filter( i -> !Modifier.isNative( i.getModifiers() ) )
290  .filter( i -> !Modifier.isStatic( i.getModifiers() ) )
291  .filter( l_filter ),
292  methods( p_class.getSuperclass() )
293  );
294  }

302  @Nonnull
303  private static Pair<Boolean, IAgentAction.EAccess> isActionClass( final Class<?> p_class )
304  {
305  if ( !p_class.isAnnotationPresent( IAgentAction.class ) )
306  return new ImmutablePair<>( false, IAgentAction.EAccess.BLACKLIST );
307 
308  final IAgentAction l_annotation = p_class.getAnnotation( IAgentAction.class );
309  return new ImmutablePair<>(
310  ( l_annotation.classes().length == 0 )
311  || ( Arrays.stream( p_class.getAnnotation( IAgentAction.class ).classes() )
312  .parallel()
313  .anyMatch( p_class::equals )
314  ),
315  l_annotation.access()
