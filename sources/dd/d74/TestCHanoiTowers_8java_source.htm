138  private void setup( final int p_agentnumber, final long p_towernumber, final long p_slicenumber,
139  final String p_asl, final Stream<IAction> p_action ) throws Exception
140  {
141  m_towernumber = p_towernumber;
142  m_slicenumber = p_slicenumber;
143  m_running = new AtomicBoolean( true );
144 
145  final Map<Integer, CTower> l_towermap = new ConcurrentHashMap<>();
146  IntStream.range( 0, m_towernumber.intValue() )
147  .forEach( i ->
148  {
149  final CTower l_tower = new CTower();
150  l_towermap.put( i, l_tower );
151  if ( i == 0 )
152  IntStream.range( 0, m_slicenumber.intValue() ).forEach( j -> l_tower.push( new CSlice( m_slicenumber.intValue() - j ) ) );
153  } );
154  m_tower = Collections.unmodifiableMap( l_towermap );
155 
156 
157  final Map<Integer, CAgent> l_agentmap = new ConcurrentHashMap<>();
158  try
159  (
160  final InputStream l_asl = new FileInputStream( p_asl )
161  )
162  {
163  final CGenerator l_generator = new CGenerator( l_asl, p_action );
164  IntStream.range( 0, p_agentnumber )
165  .forEach( i -> l_agentmap.put( i, l_generator.generatesingle( i ) ) );
166  }
167  catch ( final IOException l_exception )
168  {
169  l_exception.printStackTrace();
170  assertTrue( "asl could not be read", true );
171  }
172  m_agents = Collections.unmodifiableMap( l_agentmap );
